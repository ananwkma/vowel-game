**renderTiles(word):** Clear #tiles-container, create one .tile div per letter. Each tile:
- Sets `tile.textContent = letter` (uppercase)
- Sets `data-index` attribute

```javascript
const tileElements = [];

function renderTiles(word) {
  tilesContainer.innerHTML = '';
  tileElements.length = 0;
  const letters = word.toUpperCase().split('');

  letters.forEach((letter, idx) => {
    const tile = document.createElement('div');
    tile.className = 'tile';
    tile.setAttribute('data-index', idx);
    tile.setAttribute('role', 'button');
    tile.setAttribute('aria-label', `Tile ${idx + 1}: ${letter}`);

    // Show pending letter if this tile is being edited
    if (idx === gameState.pendingTileIndex && gameState.pendingLetter) {
      tile.textContent = gameState.pendingLetter;
      tile.classList.add('selected');
    } else {
      tile.textContent = letter;
      if (idx === gameState.pendingTileIndex) tile.classList.add('selected');
    }

    tile.addEventListener('click', () => selectTile(idx));
    tilesContainer.appendChild(tile);
    tileElements[idx] = tile;
  });
}
```

**selectTile(index):** Enforce single-pending-tile constraint:
```javascript
function selectTile(index) {
  if (gameState.completed || gameState.gaveUp) return;

  // Reset the previously pending tile to its original letter
  if (gameState.pendingTileIndex !== null && gameState.pendingTileIndex !== index) {
    tileElements[gameState.pendingTileIndex].textContent =
      gameState.currentWord[gameState.pendingTileIndex].toUpperCase();
    tileElements[gameState.pendingTileIndex].classList.remove('selected');
    gameState.pendingLetter = null;
  }

  gameState.pendingTileIndex = index;
  tileElements[index].classList.add('selected');

  // Focus hidden input so mobile keyboard opens
  hiddenInput.value = '';
  hiddenInput.focus();

  clearError();
}
```

**submitWord function:**
```javascript
async function submitWord() {
  if (gameState.completed || gameState.gaveUp) return;
  if (gameState.pendingTileIndex === null) {
    showError('Tap a tile to change a letter');
    return;
  }

  // Build the candidate word from current tiles
  const letters = gameState.currentWord.toUpperCase().split('');
  if (gameState.pendingLetter) {
    letters[gameState.pendingTileIndex] = gameState.pendingLetter;
  }
  const candidate = letters.join('');

  // Validation: count diffs
  const diffs = countLetterDiffs(gameState.currentWord, candidate);

  if (diffs === 0) {
    showError('Change exactly one letter');
    shakeTiles();
    return;
  }

  if (diffs > 1) {
    showError('Change exactly one letter');
    shakeTiles();
    return;
  }

  // Validation: dictionary check (must be a neighbor in adjacency map)
  if (!isValidNextWord(gameState.currentWord, candidate)) {
    showError('Not a valid word');
    shakeTiles();
    return;
  }

  // Valid word — proceed with stamp animation
  clearError();
  await stampAnimation();

  // Update state
  const prevWord = gameState.currentWord;
  gameState.currentWord = candidate;
  gameState.pendingTileIndex = null;
  gameState.pendingLetter = null;
  gameState.playerPath.push(candidate);

  // Add previous word to history (the "stamp" leaves this mark)
  // Note: history already has start word; subsequent words added here
  addToHistory(prevWord);

  // Check win condition (exact match)
  if (candidate === gameState.targetWord) {
    gameState.completed = true;
    DailyStatus.markCompleted('ladder');
    // Slight delay before showing results to let stamp animation settle
    setTimeout(() => showResults({ solved: true }), 400);
    return;
  }

  // Render fresh tiles for the new current word
  renderTiles(candidate);
}
```

```javascript
function stampAnimation() {
  return new Promise(resolve => {
    tilesContainer.classList.add('tiles-popping');
    tilesContainer.addEventListener('animationend', () => {
      tilesContainer.classList.remove('tiles-popping');
      resolve();
    }, { once: true });
  });
}
```

**addToHistory(word, isStart = false):**
```javascript
function addToHistory(word, isStart = false) {
  const entry = document.createElement('div');
  entry.className = 'history-entry';
  if (isStart) entry.classList.add('history-start');
  entry.textContent = word.toUpperCase();
  entry.setAttribute('aria-label', word);
  historyContainer.appendChild(entry);

  // Auto-scroll so newest entry is always visible
  historyContainer.scrollTop = historyContainer.scrollHeight;
}
```

Use pointer events (unified touch+mouse per STATE.md decision):
```javascript
function setupGiveUp() {
  const HOLD_DURATION_MS = 3000;
  let holdState = { active: false, startTime: null, timerId: null };

  giveUpBtn.addEventListener('pointerdown', (e) => {
    e.preventDefault();
    if (gameState.completed || gameState.gaveUp) return;

    holdState.active = true;
    holdState.startTime = Date.now();
    giveUpBar.style.width = '0%';

    holdState.timerId = setInterval(() => {
      const elapsed = Date.now() - holdState.startTime;
      const progress = Math.min(elapsed / HOLD_DURATION_MS, 1);
      giveUpBar.style.width = (progress * 100) + '%';

      if (progress >= 1) {
        clearInterval(holdState.timerId);
        holdState.active = false;
        triggerGiveUp();
      }
    }, 16); // ~60fps
  });

  // Cancel on pointer release (anywhere on document, in case user drags off button)
  document.addEventListener('pointerup', () => {
    if (holdState.active) {
      clearInterval(holdState.timerId);
      giveUpBar.style.width = '0%';
      holdState.active = false;
    }
  });

  document.addEventListener('pointercancel', () => {
    if (holdState.active) {
      clearInterval(holdState.timerId);
      giveUpBar.style.width = '0%';
      holdState.active = false;
    }
  });
}
```
