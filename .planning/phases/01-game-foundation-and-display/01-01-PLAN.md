Inside game.html's SECTION 2 (WORD ENGINE), implement the WordEngine object. This replaces the `/* ... WordEngine object ... */` placeholder comment.

```javascript
const WordEngine = {
  // Reference the pre-curated word list
  wordList: WORDS,

  // Track recently used words to prevent immediate repeats
  // Keep last 20 to ensure good variety across sessions
  recentWords: new Set(),
  maxRecent: 20,

  /**
   * Returns a random word not seen in the recent 20 picks.
   * Uses rejection sampling (fast given 1000+ word pool).
   */
  getRandomWord() {
    // Safety: if pool is smaller than maxRecent, shrink window
    const window = Math.min(this.maxRecent, Math.floor(this.wordList.length / 2));
    let word;
    let attempts = 0;
    do {
      const idx = Math.floor(Math.random() * this.wordList.length);
      word = this.wordList[idx];
      attempts++;
      // Escape hatch: if somehow stuck (tiny list), clear recent
      if (attempts > 100) {
        this.recentWords.clear();
        break;
      }
    } while (this.recentWords.has(word));

    // Track and evict oldest if over window size
    this.recentWords.add(word);
    if (this.recentWords.size > window) {
      const oldest = this.recentWords.values().next().value;
      this.recentWords.delete(oldest);
    }

    return word;
  },

  /**
   * Extracts consonants from word in their original order.
   * Explicitly uses consonant character class [b-df-hj-np-tv-z]
   * to correctly exclude Y (which can be a vowel) and A/E/I/O/U.
   * Returns uppercase array.
   */
  extractConsonants(word) {
    return (word.toUpperCase().match(/[B-DF-HJ-NP-TV-Z]/g) || []);
  },

  /**
   * Extracts vowels from word in their original order.
   * Only A, E, I, O, U — no Y.
   * Returns uppercase array.
   */
  extractVowels(word) {
    return (word.toUpperCase().match(/[AEIOU]/g) || []);
  },

  /**
   * Returns true if word qualifies as a puzzle word:
   * - 4-10 letters long
   * - Contains at least one A/E/I/O/U vowel
   */
  isValidGameWord(word) {
    const upper = word.toUpperCase();
    return upper.length >= 4 && upper.length <= 10 && /[AEIOU]/.test(upper);
  },
};
```
