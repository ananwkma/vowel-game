**Replace stub functions in the script block with these implementations:**

```javascript
let selectedNumber = null;

function selectBlock(number) {
  if (gameState.won) return;

  // Deselect all previously selected blocks
  document.querySelectorAll('.cipher-block.selected').forEach(b => b.classList.remove('selected'));

  selectedNumber = number;

  // Highlight ALL blocks with this number (amber outline)
  document.querySelectorAll(`.cipher-block[data-number="${number}"]`).forEach(b => {
    b.classList.add('selected');
  });

  // Focus hidden input to open mobile keyboard
  const hiddenInput = document.getElementById('hidden-input');
  hiddenInput.value = '';
  hiddenInput.focus();
}

function assignLetter(number, letter) {
  // Duplicate enforcement: if this letter is already assigned to another number, clear it
  Object.entries(guesses).forEach(([existingNum, existingLetter]) => {
    if (existingLetter === letter && parseInt(existingNum) !== number) {
      const n = parseInt(existingNum);
      delete guesses[n];
      document.querySelectorAll(`.cipher-block[data-number="${n}"]`).forEach(block => {
        block.textContent = n;
        block.classList.remove('correct', 'wrong', 'selected');
        block.classList.add('unsolved');
      });
    }
  });

  // Assign letter to this number
  guesses[number] = letter;
  const isCorrect = cipherMap[number] === letter;

  document.querySelectorAll(`.cipher-block[data-number="${number}"]`).forEach(block => {
    block.textContent = letter;
    block.classList.remove('unsolved', 'correct', 'wrong', 'selected');
    block.classList.add(isCorrect ? 'correct' : 'wrong');
  });

  selectedNumber = null;

  // Save progress after every assignment
  CipherProgress.save(guesses);

  // Check win
  checkWin();
}

function checkWin() {
  const totalNumbers = Object.keys(cipherMap).length;
  const solvedCount = Object.keys(cipherMap).filter(
    num => guesses[parseInt(num)] === cipherMap[parseInt(num)]
  ).length;
  if (solvedCount === totalNumbers) {
    gameState.won = true;
    DailyStatus.markCompleted('cipher');
    showWinScreen();
  }
}

function showWinScreen() {
  // Show confetti
  const canvas = document.getElementById('confetti-canvas');
  canvas.style.display = 'block';
  runConfetti(canvas, 3000);  // 3 seconds of confetti

  // Populate and show win overlay
  document.getElementById('win-quote').textContent = QUOTE_TEXT;
  document.getElementById('win-author-name').textContent = QUOTE_AUTHOR;
  const overlay = document.getElementById('win-overlay');
  overlay.style.display = 'flex';
}

// Confetti — project palette colors, canvas-based (same approach as ladder.html)
function runConfetti(canvas, duration) {
  const ctx = canvas.getContext('2d');
  canvas.width = window.innerWidth;
  canvas.height = window.innerHeight;
  const particles = [];
  const colors = ['#D4A574', '#8BAF7C', '#C4836F', '#2C2B28', '#F8F7F4'];
  const count = 80;

  for (let i = 0; i < count; i++) {
    particles.push({
      x: Math.random() * canvas.width,
      y: Math.random() * canvas.height - canvas.height,
      w: Math.random() * 10 + 6,
      h: Math.random() * 5 + 4,
      color: colors[Math.floor(Math.random() * colors.length)],
      speed: Math.random() * 3 + 2,
      angle: Math.random() * Math.PI * 2,
      spin: (Math.random() - 0.5) * 0.15,
      drift: (Math.random() - 0.5) * 1.5
    });
  }

  const end = Date.now() + duration;
  function frame() {
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    particles.forEach(p => {
      p.y += p.speed;
      p.x += p.drift;
      p.angle += p.spin;
      ctx.save();
      ctx.translate(p.x + p.w / 2, p.y + p.h / 2);
      ctx.rotate(p.angle);
      ctx.fillStyle = p.color;
      ctx.fillRect(-p.w / 2, -p.h / 2, p.w, p.h);
      ctx.restore();
    });
    if (Date.now() < end) {
      requestAnimationFrame(frame);
    } else {
      canvas.style.display = 'none';
    }
  }
  requestAnimationFrame(frame);
}
```
