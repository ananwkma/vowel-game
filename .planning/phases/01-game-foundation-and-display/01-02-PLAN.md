**SECTION 4: DOM RENDERING**

```javascript
/* ============================================================
   SECTION 4: DOM RENDERING
   renderBoard(state) — pure function converting state to DOM.
   Always clear and re-render; never patch.
============================================================ */
function renderBoard(state) {
  const board = document.getElementById('game-board');
  board.innerHTML = ''; // Clear previous render

  // Create a document fragment for efficient batch DOM update
  const fragment = document.createDocumentFragment();

  // 1. Render blank vowel placeholder blocks on the LEFT
  //    (one per vowel in the original word)
  for (let i = 0; i < state.vowelCount; i++) {
    const block = document.createElement('div');
    block.className = 'block vowel-block empty';
    block.textContent = '_';
    block.setAttribute('data-vowel-index', i);
    block.setAttribute('aria-label', 'Vowel placeholder ' + (i + 1));
    block.setAttribute('role', 'button');
    block.setAttribute('tabindex', '0');
    fragment.appendChild(block);
  }

  // 2. Render consonant blocks on the RIGHT, in original order
  state.consonants.forEach((letter, idx) => {
    const block = document.createElement('div');
    block.className = 'block consonant-block';
    block.textContent = letter;
    block.setAttribute('data-consonant-index', idx);
    block.setAttribute('aria-label', 'Consonant ' + letter);
    block.setAttribute('aria-disabled', 'true');
    fragment.appendChild(block);
  });

  board.appendChild(fragment);
}
```

**SECTION 6: INIT**

```javascript
/* ============================================================
   SECTION 6: INIT
   initGame() — called once on load and after each round ends.
   Selects a word, extracts letters, updates state, renders board.
============================================================ */
function initGame() {
  const word = WordEngine.getRandomWord();

  gameState.currentWord = word;
  gameState.consonants = WordEngine.extractConsonants(word);
  gameState.vowels = WordEngine.extractVowels(word);
  gameState.vowelCount = gameState.vowels.length;
  gameState.phase = 'playing';

  renderBoard(gameState);

  console.log('[Game] New word loaded:', word,
    '| Consonants:', gameState.consonants.join(''),
    '| Vowel count:', gameState.vowelCount);
}

// Start the game when the DOM is ready
document.addEventListener('DOMContentLoaded', initGame);
```
