**`setupDragSelection()` function:**

```javascript
function setupDragSelection() {
  const gridContainer = document.getElementById('grid-container');

  gridContainer.addEventListener('pointerdown', (e) => {
    if (gameState.phase === 'done') return;
    e.preventDefault();
    isDragging = true;
    selectedCells = [];
    gridContainer.setPointerCapture(e.pointerId); // keep tracking outside element bounds
    const cell = getCellAt(e.clientX, e.clientY);
    if (cell) addCellToSelection(cell);
  });

  gridContainer.addEventListener('pointermove', (e) => {
    if (!isDragging) return;
    e.preventDefault();
    const cell = getCellAt(e.clientX, e.clientY);
    if (cell && !isCellInSelection(cell) && isAdjacentToLast(cell)) {
      addCellToSelection(cell);
      drawTraceLine();
    }
  });

  gridContainer.addEventListener('pointerup', () => {
    if (!isDragging) return;
    isDragging = false;
    evaluateSelection();
  });

  gridContainer.addEventListener('pointercancel', () => {
    isDragging = false;
    clearSelection();
  });
}
```

**`getCellAt(clientX, clientY)`:**
```javascript
function getCellAt(clientX, clientY) {
  // Canvas has pointer-events:none so elementFromPoint hits the cell underneath
  const el = document.elementFromPoint(clientX, clientY);
  if (el && el.classList.contains('grid-cell')) {
    return { row: parseInt(el.dataset.row), col: parseInt(el.dataset.col), el };
  }
  return null;
}
```

**`isAdjacentToLast(cell)`:** Returns true only if cell is one of the 8 neighbors of the last selected cell (prevents diagonal skipping on budget devices — Pitfall 7). If selectedCells is empty, always returns true.

**`isCellInSelection(cell)`:** Returns true if `selectedCells` already contains a cell with the same row and col.

**`addCellToSelection(cell)`:** Push to selectedCells, add `active` class to cell.el.

**`clearSelection()`:** Remove `active` class from all non-found cells in selectedCells; reset selectedCells = []; call clearCanvas().

function evaluateSelection() {
  if (selectedCells.length === 0) return;
  const word = selectedCells.map(c => gridData[c.row][c.col]).join('');
  const unfound = dailyPuzzle.words.filter(w =>
    !gameState.wordsFound.find(f => f.word === w.word)

    markWordFound(match, false); // false = no hint used

}

**`useHint()`:** Called by hint button click handler.
```javascript
function useHint() {
  if (gameState.phase === 'done') return;

  // Find unfound words for current phase
  const currentPhase = gameState.phase; // 'easy' or 'hard'
  const unfound = dailyPuzzle.words.filter(w =>
    w.phase === currentPhase &&
    !gameState.wordsFound.find(f => f.word === w.word)
  );
  if (unfound.length === 0) return;

  // Round-robin through unfound words
  const targetWord = unfound[gameState.hintIndex % unfound.length];
  gameState.hintIndex++;

  // Find the placement for this word
  const placement = wordPlacements.find(p => p.word === targetWord.word);
  if (!placement) return;

  // Pulse the first cell of this word (amber highlight for 1.5s)
  const firstCell = placement.cells[0];
  const el = document.querySelector(`[data-row="${firstCell.row}"][data-col="${firstCell.col}"]`);
  if (el && !el.classList.contains('found') && !el.classList.contains('found-hint')) {
    el.classList.add('active');
    setTimeout(() => el.classList.remove('active'), 1500);
  }

  // Add 30s penalty to current phase timer
  const HINT_PENALTY_MS = 30000; // 30 seconds per hint
  if (currentPhase === 'easy') {
    gameState.hintPenaltyEasy += HINT_PENALTY_MS;
  } else {
    gameState.hintPenaltyHard += HINT_PENALTY_MS;
  }

  // Mark this word as hint-assisted (even if player finds it themselves later,
  // treat as hinted once a hint for it was used — simple model)
  // Actually: only mark as hinted when found (handled in markWordFound).
  // Store which words have had hints used in a Set for pip tracking:
  if (!gameState.hintedWords) gameState.hintedWords = new Set();
  gameState.hintedWords.add(targetWord.word);
}
```

**Give-up implementation (per RESEARCH.md Pitfall 9 / CONTEXT.md discretion):**
Use a two-tap confirm pattern:
```javascript
let giveUpPending = false;
let giveUpTimer = null;

function handleGiveUp() {
  if (gameState.phase === 'done') return;
  if (!giveUpPending) {
    giveUpPending = true;
    document.getElementById('btn-giveup').textContent = 'Sure?';
    giveUpTimer = setTimeout(() => {
      giveUpPending = false;
      document.getElementById('btn-giveup').textContent = 'Give Up';
    }, 3000);
    return;
  }
  // Second tap: confirm
  clearTimeout(giveUpTimer);
  giveUpPending = false;
  // Reveal all unfound words (highlight in warning/rose)
  dailyPuzzle.words.forEach(w => {
    if (!gameState.wordsFound.find(f => f.word === w.word)) {
      const placement = wordPlacements.find(p => p.word === w.word);
      if (placement) {
        placement.cells.forEach(({row, col}) => {
          const el = document.querySelector(`[data-row="${row}"][data-col="${col}"]`);
          if (el) { el.classList.remove('active'); el.classList.add('found-hint'); }
        });
        gameState.wordsFound.push({ word: w.word, usedHint: true });
      }
    }
  });
  // Stop hard phase timer if running (easy already stopped at reveal)
  if (gameState.phase === 'hard' && gameState.hardTimerInterval) stopHardTimer();
  else if (gameState.phase === 'easy') stopEasyTimer();
  renderPips();
  endGame(true); // gaveUp = true
}
```
