**animateOptimalPath function (called on give-up, before showResults):**

Animates OPTIMAL_PATH word-by-word with ~300ms stagger. Adds words directly into the results screen's `.results-optimal-path` container so they're visible when the screen appears.

```javascript
async function animateOptimalPath(container) {
  // container is the .results-optimal-path div, already in the DOM but results screen not yet visible
  if (!OPTIMAL_PATH) return;

  for (let i = 0; i < OPTIMAL_PATH.length; i++) {
    const word = OPTIMAL_PATH[i];
    const el = document.createElement('div');
    el.className = 'optimal-word';
    el.textContent = word;

    // Mark start and target for visual distinction
    if (i === 0) el.classList.add('optimal-start');
    if (i === OPTIMAL_PATH.length - 1) el.classList.add('optimal-end');

    container.appendChild(el);

    // Stagger each word reveal: append hidden, then fade in after brief delay
    await new Promise(resolve => setTimeout(resolve, 300));
    el.classList.add('revealed');
  }
}
```

async function showResults({ solved }) {
  const resultsScreen = document.getElementById('results-screen');
  const steps = gameState.playerPath.length - 1; // Steps taken (start not counted)
  const delta = steps - OPTIMAL_STEPS;          // How far from optimal

  // Build results screen HTML
  let titleText, subtitleText, extraContent = '', buttonRow = '';

  if (solved) {
    if (delta === 0) {
      titleText = 'Optimal!';
      subtitleText = `Solved in ${steps} step${steps !== 1 ? 's' : ''} — perfect solution`;
    } else {
      titleText = 'Solved!';
      subtitleText = `Solved in ${steps} step${steps !== 1 ? 's' : ''} (+${delta} from optimal)`;
    }

    // Show optimal path (non-animated — it's revealed immediately since player solved)
    if (OPTIMAL_PATH) {
      extraContent = `
        <div class="results-section-label">Optimal path (${OPTIMAL_STEPS} step${OPTIMAL_STEPS !== 1 ? 's' : ''})</div>
        <div class="results-optimal-path" id="results-optimal-path">
          ${OPTIMAL_PATH.map((w, i) => `<div class="optimal-word revealed ${i === 0 ? 'optimal-start' : ''} ${i === OPTIMAL_PATH.length - 1 ? 'optimal-end' : ''}">${w}</div>`).join('')}
        </div>`;
    }

    if (delta > 0) {
      buttonRow += `<button class="results-btn btn-secondary" id="try-again-btn" type="button">Try Again</button>`;
    }
  } else {
    // Gave up
    titleText = 'Gave Up';
    const remaining = gameState.playerPath.length > 1
      ? `after ${gameState.playerPath.length - 1} step${gameState.playerPath.length - 1 !== 1 ? 's' : ''}`
      : 'at the start';
    subtitleText = `${remaining} — optimal path was ${OPTIMAL_STEPS} step${OPTIMAL_STEPS !== 1 ? 's' : ''}`;

    // Optimal path will be animated — add empty container
    extraContent = `
      <div class="results-section-label">Optimal path (${OPTIMAL_STEPS} step${OPTIMAL_STEPS !== 1 ? 's' : ''})</div>
      <div class="results-optimal-path" id="results-optimal-path"></div>`;
  }

  buttonRow += `
    <button class="results-btn btn-primary" id="share-btn" type="button">Share</button>
    <a href="index.html" class="results-btn btn-secondary" style="text-decoration:none;display:inline-flex;align-items:center;justify-content:center;">← Word Games</a>`;

  resultsScreen.innerHTML = `
    <div class="results-title">${titleText}</div>
    <div class="results-steps">${subtitleText}</div>
    ${extraContent}
    <div class="results-actions">${buttonRow}</div>`;

  // Add .results-actions CSS: display flex, flex-wrap wrap, gap var(--spacing-sm), justify-content center, margin-top var(--spacing-lg)

  // Show results screen
  resultsScreen.hidden = false;

  // Wire Try Again if present (reset game without page reload)
  const tryAgainBtn = resultsScreen.querySelector('#try-again-btn');
  if (tryAgainBtn) {
    tryAgainBtn.addEventListener('click', resetGame);
  }

  // Wire Share button
  const shareBtn = resultsScreen.querySelector('#share-btn');
  if (shareBtn) {
    shareBtn.addEventListener('click', () => shareResult({ solved, steps, delta }));
  }

  // If give-up: animate optimal path after showing screen
  if (!solved) {
    const optimalContainer = resultsScreen.querySelector('#results-optimal-path');
    await animateOptimalPath(optimalContainer);
  }

  // If optimal solve: fire confetti
  if (solved && delta === 0) {
    fireConfetti();
  }
}

**shareResult({ solved, steps, delta }):**
```javascript
function shareResult({ solved, steps, delta }) {
  const today = new Date().toISOString().split('T')[0];
  const start = dailyPuzzle.start;
  const target = dailyPuzzle.target;

  let text;
  if (solved) {
    if (delta === 0) {
      text = `Word Ladder ${today} | ${start} → ${target} | Solved in ${steps} step${steps !== 1 ? 's' : ''} (Optimal!) 🎯`;
    } else {
      text = `Word Ladder ${today} | ${start} → ${target} | Solved in ${steps} step${steps !== 1 ? 's' : ''} (+${delta} from optimal)`;
    }
  } else {
    text = `Word Ladder ${today} | ${start} → ${target} | Gave up | Optimal: ${OPTIMAL_STEPS} step${OPTIMAL_STEPS !== 1 ? 's' : ''}`;
  }

  if (navigator.clipboard && navigator.clipboard.writeText) {
    navigator.clipboard.writeText(text).then(() => {
      // Brief visual confirmation on the share button
      const shareBtn = document.getElementById('share-btn');
      if (shareBtn) {
        const original = shareBtn.textContent;
        shareBtn.textContent = 'Copied!';
        setTimeout(() => { shareBtn.textContent = original; }, 1500);
      }
    }).catch(() => {
      // Fallback: prompt with text
      prompt('Copy this result:', text);
    });
  } else {
    prompt('Copy this result:', text);
  }
}
```

**resetGame (Try Again — resets state, re-renders tiles, hides results screen):**
```javascript
function resetGame() {
  // Reset game state — same daily puzzle, fresh attempt
  gameState.currentWord = dailyPuzzle.start;
  gameState.pendingTileIndex = null;
  gameState.pendingLetter = null;
  gameState.playerPath = [dailyPuzzle.start];
  gameState.completed = false;
  gameState.gaveUp = false;

  // Clear UI
  historyContainer.innerHTML = '';
  clearError();
  giveUpBar.style.width = '0%';

  // Re-initialize game UI
  addToHistory(dailyPuzzle.start, true); // Start word back as first history entry
  renderTiles(dailyPuzzle.start);

  // Hide results screen
  const resultsScreen = document.getElementById('results-screen');
  resultsScreen.hidden = true;
  resultsScreen.innerHTML = '';
}
```
