**Step 2: Build adjacency map at startup.**
```javascript
function buildAdjacencyMap(words) {
  const adjacency = {};
  const byLength = {};

  words.forEach(word => {
    const upper = word.toUpperCase();
    const len = upper.length;
    if (!byLength[len]) byLength[len] = [];
    byLength[len].push(upper);
    adjacency[upper] = []; // Initialize all words
  });

  words.forEach(word => {
    const upper = word.toUpperCase();
    const candidates = byLength[upper.length] || [];
    candidates.forEach(candidate => {
      if (candidate === upper) return;
      let diffs = 0;
      for (let i = 0; i < upper.length; i++) {
        if (upper[i] !== candidate[i]) diffs++;
        if (diffs > 1) break; // Early exit
      }
      if (diffs === 1) adjacency[upper].push(candidate);
    });
  });

  return adjacency;
}

const ADJACENCY = buildAdjacencyMap(DICTIONARY);
```

**Step 3: BFS shortest path function.**
```javascript
function bfsShortestPath(start, target, adjacency) {
  const startU = start.toUpperCase();
  const targetU = target.toUpperCase();
  if (startU === targetU) return [startU];

  const visited = new Set([startU]);
  const queue = [[startU, [startU]]];

  while (queue.length > 0) {
    const [current, path] = queue.shift();
    const neighbors = adjacency[current] || [];
    for (const neighbor of neighbors) {
      if (neighbor === targetU) return [...path, targetU];
      if (!visited.has(neighbor)) {
        visited.add(neighbor);
        queue.push([neighbor, [...path, neighbor]]);
      }
    }
  }
  return null;
}
```

**Step 4: Seeded PRNG (copy exact pattern from vowel.html).**
```javascript
function seededRandom(seedStr) {
  let hash = 0;
  for (let i = 0; i < seedStr.length; i++) {
    hash = ((hash << 5) - hash) + seedStr.charCodeAt(i);
    hash = hash & hash;
  }
  const m = 2147483647;
  const a = 16807;
  let seed = Math.abs(hash) % m || 1;
  return function() {
    seed = (a * seed) % m;
    return seed / m;
  };
}
```

```javascript
const DATE_SEED = (() => {
  const today = new Date().toISOString().split('T')[0]; // YYYY-MM-DD
  return today + '_ladder_v1';
})();

function getDailyLadderPuzzle() {
  // Only consider 5-letter words (most ladder puzzles use 5-letter words)
  const fiveLetterWords = Object.keys(ADJACENCY).filter(w =>
    w.length === 5 && ADJACENCY[w].length >= 2 // Must have at least 2 neighbors
  );

  if (fiveLetterWords.length < 2) return null;

  // Try up to 20 seed offsets to find a valid pair with a BFS path
  for (let attempt = 0; attempt < 20; attempt++) {
    const rng = seededRandom(DATE_SEED + '_' + attempt);
    const startIdx = Math.floor(rng() * fiveLetterWords.length);
    const targetIdx = Math.floor(rng() * fiveLetterWords.length);

    const start = fiveLetterWords[startIdx];
    const target = fiveLetterWords[targetIdx];

    if (start === target) continue;

    const path = bfsShortestPath(start, target, ADJACENCY);
    if (path && path.length >= 4 && path.length <= 10) {
      // Require path of 3-9 steps (not trivially short, not impossibly long)
      return { start, target, optimalPath: path };
    }
  }

  // Absolute fallback: known-good pair
  return { start: 'STONE', target: 'CRANE', optimalPath: bfsShortestPath('STONE', 'CRANE', ADJACENCY) };
}

const dailyPuzzle = getDailyLadderPuzzle();
const OPTIMAL_PATH = dailyPuzzle.optimalPath;
const OPTIMAL_STEPS = OPTIMAL_PATH ? OPTIMAL_PATH.length - 1 : 0;
```

```javascript
function countLetterDiffs(wordA, wordB) {
  const a = wordA.toUpperCase();
  const b = wordB.toUpperCase();
  if (a.length !== b.length) return Infinity;
  let diffs = 0;
  for (let i = 0; i < a.length; i++) {
    if (a[i] !== b[i]) diffs++;
  }
  return diffs;
}

function isValidNextWord(currentWord, candidateWord) {
```
