Add a new JS section immediately before SECTION 6 (initGame) in the first `<script>` block. Name it `SECTION 5.8: ELAPSED TIMER`. Implement `elapsedTimer` as a plain object:

```javascript
/* ============================================================
   SECTION 5.8: ELAPSED TIMER
   Counts up from puzzleState.timerElapsed (restored on reload).
   Displays M:SS format. Runs continuously through word transitions.
   Stops when showPuzzleComplete() is called.
============================================================ */
const elapsedTimer = (function() {
  let intervalId = null;
  let baseSeconds = 0;   // accumulated seconds from previous words + restored from localStorage
  let startTime = null;  // Date.now() at last start(), for drift-free counting

  const el = document.getElementById('elapsed-timer');

  function formatTime(totalSeconds) {
    const m = Math.floor(totalSeconds / 60);
    const s = totalSeconds % 60;
    return m + ':' + (s < 10 ? '0' : '') + s;
  }

  function tick() {
    const elapsed = baseSeconds + Math.floor((Date.now() - startTime) / 1000);
    if (el) el.textContent = formatTime(elapsed);
    // Sync puzzleState so savePuzzleState() always has the latest value
    puzzleState.timerElapsed = elapsed;
  }

  function start() {
    if (intervalId !== null) return; // already running — do not reset
    startTime = Date.now();
    // Render immediately so there's no 1-second blank delay
    tick();
    intervalId = setInterval(tick, 1000);
  }

  function stop() {
    if (intervalId !== null) {
      clearInterval(intervalId);
      intervalId = null;
    }
    // Capture final value into puzzleState
    if (startTime !== null) {
      puzzleState.timerElapsed = baseSeconds + Math.floor((Date.now() - startTime) / 1000);
    }
    if (el) el.textContent = formatTime(puzzleState.timerElapsed);
  }

  // Called after loadPuzzleState() to restore base from persisted timerElapsed
  function setBase(seconds) {
    baseSeconds = typeof seconds === 'number' ? seconds : 0;
    if (el) el.textContent = formatTime(baseSeconds);
  }

  return { start, stop, setBase };
})();
```

This must be inserted in BOTH code paths:
