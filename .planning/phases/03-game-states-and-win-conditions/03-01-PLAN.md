**Step 2 — Add WordSet and validation functions** in a new SECTION 7 comment block, placed after the drag logic block but before the DOMContentLoaded listener:

```javascript
/* ============================================================
   SECTION 7: VALIDATION & STATE MANAGEMENT
============================================================ */

// O(1) word membership lookup — built once at startup from the WORDS array
const WordSet = new Set(WORDS);

/**
 * Returns the word constructed from blocks in their current DOM order.
 * Reads left-to-right visual order from the game board.
 * Skips placeholder-block elements (drag artifacts).
 * Returns uppercase string.
 */
function constructedWord() {
  const board = document.getElementById('game-board');
  const blocks = Array.from(board.children).filter(
    el => el.classList.contains('block') &&
          !el.classList.contains('placeholder-block') &&
          !el.classList.contains('vowel-guide')
  );
  return blocks.map(b => b.textContent.trim()).join('').toUpperCase();
}

/**
 * Checks win condition:
 * - Game must be in 'playing' phase (guard against re-triggering)
 * - All vowel-blocks must be present in the DOM (count matches vowelCount)
 * - All vowel-blocks must show a vowel letter (not '_')
 * - The constructed word must exist in WordSet
 * Returns true if win triggered, false otherwise.
 */
function checkWinCondition() {
  if (gameState.phase !== 'playing') return false;

  const board = document.getElementById('game-board');
  const vowelBlocks = Array.from(board.querySelectorAll('.vowel-block:not(.placeholder-block)'));

  const allPlaced = vowelBlocks.length >= gameState.vowelCount;
  const allHaveVowels = allPlaced && vowelBlocks.every(b => b.textContent.trim() !== '_');

  if (allPlaced && allHaveVowels) {
    const word = constructedWord();
    console.log('[Validation] Checking word:', word, '| Valid:', WordSet.has(word));
    if (WordSet.has(word)) {
      gameState.phase = 'won';
      applyStateVisuals('won');
      scheduleAutoAdvance(2000);
      return true;
    }
  }
  return false;
}
```

**Step 3 — Add visual feedback and auto-advance functions** immediately after the validation block, still in SECTION 7:

```javascript
/* ============================================================
   SECTION 8: ANIMATION & TIMING
============================================================ */

// Single timer reference — prevents overlapping auto-advances
let advanceTimer = null;

/**
 * Applies background color for the current game phase.
 * Uses CSS transition on body for smooth color change.
 */
function applyStateVisuals(phase) {
  const body = document.body;
  if (phase === 'won') {
    body.style.transition = 'background-color 0.3s ease';
    body.style.backgroundColor = '#8BAF7C'; // Sage green (--color-win-bg)
  } else if (phase === 'gaveup') {
    body.style.transition = 'background-color 0.3s ease';
    body.style.backgroundColor = '#C4836F'; // Dusty rose (--color-lose-bg)
  } else {
    body.style.transition = 'background-color 0.5s ease';
    body.style.backgroundColor = '#F8F7F4'; // Off-white (--color-bg)
  }
}

/**
 * Schedules auto-advance to next puzzle after delayMs.
 * Clears any previously scheduled advance before setting new one.
 */
function scheduleAutoAdvance(delayMs) {
  if (advanceTimer) clearTimeout(advanceTimer);
  advanceTimer = setTimeout(() => {
    applyStateVisuals('playing');
    initGame();
    advanceTimer = null;
  }, delayMs);
}
```
